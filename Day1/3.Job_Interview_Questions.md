### Q2. Explain TypeScript Generics with a real use case

**Answer:**
Generics allow you to write **reusable, type-safe** code that works with multiple types.

```typescript
// Generic API response wrapper
interface ApiResponse<T> {
  data: T
  status: number
  message: string
  timestamp: Date
}

// Generic fetch utility
async function fetchData<T>(url: string): Promise<ApiResponse<T>> {
  const res = await fetch(url)
  return res.json()
}

// Usage â€” TypeScript infers the type
const users = await fetchData<User[]>('/api/users')
// users.data is typed as User[] âœ…

const product = await fetchData<Product>('/api/products/1')
// product.data is typed as Product âœ…
```

**Generic Hook Example:**
```tsx
// hooks/useFetch.ts
function useFetch<T>(url: string) {
  const [data, setData] = useState<T | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<Error | null>(null)

  useEffect(() => {
    fetch(url)
      .then(r => r.json())
      .then((d: T) => setData(d))
      .catch(setError)
      .finally(() => setLoading(false))
  }, [url])

  return { data, loading, error }
}

// Usage
const { data: users } = useFetch<User[]>('/api/users')
```

### Q5. How do you handle Custom Hooks in React with TypeScript?

```tsx
// hooks/useLocalStorage.ts
function useLocalStorage<T>(key: string, initialValue: T) {
  const [storedValue, setStoredValue] = useState<T>(() => {
    try {
      const item = window.localStorage.getItem(key)
      return item ? (JSON.parse(item) as T) : initialValue
    } catch {
      return initialValue
    }
  })

  const setValue = (value: T | ((val: T) => T)) => {
    const valueToStore = value instanceof Function ? value(storedValue) : value
    setStoredValue(valueToStore)
    window.localStorage.setItem(key, JSON.stringify(valueToStore))
  }

  return [storedValue, setValue] as const
}

// Usage
const [theme, setTheme] = useLocalStorage<'light' | 'dark'>('theme', 'light')
```

### Q7. Explain Go Goroutines and Channels â€” when would you use them?

```go
// Fetching multiple APIs concurrently
package main

import (
    "fmt"
    "sync"
)

type Result struct {
    Source string
    Data   interface{}
}

func fetchConcurrently() {
    sources := []string{"users", "products", "orders"}
    results := make(chan Result, len(sources)) // Buffered channel
    var wg sync.WaitGroup

    for _, source := range sources {
        wg.Add(1)
        go func(src string) { // Goroutine
            defer wg.Done()
            data := fetchFromAPI(src) // Simulated API call
            results <- Result{Source: src, Data: data}
        }(source)
    }

    // Close channel when all goroutines finish
    go func() {
        wg.Wait()
        close(results)
    }()

    // Collect results
    for result := range results {
        fmt.Printf("Got %s: %v\n", result.Source, result.Data)
    }
}
```

**Real-World Scenario:**
> A dashboard needs data from 5 microservices. Sequential: 5 Ã— 200ms = 1 second. With goroutines: ~200ms (all run in parallel).

### Q12. Write a complex SQL query â€” explain query optimization

```sql
-- âŒ Unoptimized â€” N+1 problem
SELECT * FROM orders;
-- Then for each order:
SELECT * FROM order_items WHERE order_id = ?;

-- âœ… Optimized â€” Single JOIN query
SELECT
    o.id AS order_id,
    o.created_at,
    u.name AS customer_name,
    u.email,
    COUNT(oi.id) AS item_count,
    SUM(oi.price * oi.quantity) AS total_amount
FROM orders o
JOIN users u ON o.user_id = u.id
JOIN order_items oi ON oi.order_id = o.id
WHERE o.status = 'completed'
  AND o.created_at > NOW() - INTERVAL '30 days'
GROUP BY o.id, u.name, u.email
HAVING SUM(oi.price * oi.quantity) > 100
ORDER BY total_amount DESC
LIMIT 50;

-- Index for optimization
CREATE INDEX idx_orders_status_date ON orders(status, created_at);
CREATE INDEX idx_order_items_order_id ON order_items(order_id);
```

### Q23. How would you integrate an LLM API into a Next.js app?

```tsx
// app/api/chat/route.ts
import { OpenAI } from 'openai'
import { StreamingTextResponse, OpenAIStream } from 'ai' // Vercel AI SDK

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY })

export async function POST(req: Request) {
  const { messages } = await req.json()

  const response = await openai.chat.completions.create({
    model: 'gpt-4o',
    stream: true, // Stream tokens as they generate
    messages: [
      { role: 'system', content: 'You are a helpful assistant.' },
      ...messages
    ]
  })

  const stream = OpenAIStream(response)
  return new StreamingTextResponse(stream) // Streams to client in real-time
}
```
